//! Memoization utilities. based on work of [@hackwaly](https://taolun.moonbitlang.com/t/topic/958/2)

///|
pub fn memoize1[K : Hash + Eq, V](f : (K) -> V) -> (K) -> V {
  let map = @hashmap.new()
  fn(k : K) -> V {
    match map[k] {
      Some(v) => v
      None => {
        let v = f(k)
        map[k] = v
        v
      }
    }
  }
}

///|
pub fn memoize2[K1 : Hash + Eq, K2 : Hash + Eq, V](
  f : (K1, K2) -> V
) -> (K1, K2) -> V {
  let map = @hashmap.new()
  fn(k1 : K1, k2 : K2) -> V {
    match map[(k1, k2)] {
      Some(v) => v
      None => {
        let v = f(k1, k2)
        map[(k1, k2)] = v
        v
      }
    }
  }
}
