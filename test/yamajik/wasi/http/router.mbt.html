///|
enum RouterNode {
  Root
  String(String)
  Variable(String)
} derive(Show)

///|
fn RouterNode::to_string(self : RouterNode) -> String {
  match self {
    RouterNode::Root => ""
    RouterNode::String(s) => s
    RouterNode::Variable(s) => s
  }
}

///|
fn RouterNode::from_string(s : String) -> RouterNode {
  if s == "" {
    RouterNode::Root
  } else if s.starts_with("{") && s.ends_with("}") {
    RouterNode::Variable(s)
  } else {
    RouterNode::String(s)
  }
}

///|
pub fn RouterNode::matches(self : RouterNode, s : String) -> Bool {
  match self {
    RouterNode::Root => s == ""
    RouterNode::String(s1) => s == s1
    RouterNode::Variable(_) => true
  }
}

///|
struct RouterTree {
  node : RouterNode
  handlers : Map[Method, Handler]
  children : Array[RouterTree]
}

///|
fn RouterTree::new(
  node~ : RouterNode = RouterNode::Root,
  handlers~ : Map[Method, Handler] = Map::new(),
  children~ : Array[RouterTree] = []
) -> RouterTree {
  { node, handlers, children }
}

///|
fn RouterTree::push(self : RouterTree, child : RouterTree) -> Unit {
  self.children.push(child)
}

///|
fn RouterTree::append(self : RouterTree, children : Array[RouterTree]) -> Unit {
  self.children.append(children)
}

///|
fn RouterTree::find(self : RouterTree, s : String) -> RouterTree? {
  self.children.iter().find_first(fn { child => child.node.matches(s) })
}

///|
fn RouterTree::get(self : RouterTree, path : String) -> RouterTree? {
  let parts = path.split("/").filter(fn { p => p != "" })
  let mut tree = self
  for part in parts {
    tree = match tree.find(part) {
      Some(child) => child
      None => return None
    }
  }
  Some(tree)
}

///|
fn RouterTree::get_or_init(self : RouterTree, path : String) -> RouterTree {
  let parts = path.split("/").filter(fn { p => p != "" })
  let mut tree = self
  for part in parts {
    tree = match tree.find(part) {
      Some(child) => child
      None => {
        let child = RouterTree::new(node=RouterNode::from_string(part))
        tree.push(child)
        child
      }
    }
  }
  tree
}

///|
fn RouterTree::set_handler(
  self : RouterTree,
  method~ : Method,
  handler~ : Handler
) -> Unit {
  self.handlers.set(method, handler)
}

///|
struct Router {
  tree : RouterTree
}

///|
fn Router::new(tree~ : RouterTree = RouterTree::new()) -> Router {
  { tree, }
}

///|
fn Router::sub(self : Router, path : String) -> Router {
  Router::new(tree=self.tree.get_or_init(path))
}

///|
pub fn Router::on_handle(
  self : Router,
  path~ : String,
  method~ : Method,
  handler~ : Handler
) -> Unit {
  self.tree.get_or_init(path).set_handler(method~, handler~)
}

///|
fn Router::get_handler(
  self : Router,
  path~ : String,
  method~ : Method
) -> Handler? {
  match self.tree.get(path) {
    Some(tree) => tree.handlers.get(method)
    None => None
  }
}
