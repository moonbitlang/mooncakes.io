///|
pub(all) enum Header {
  ContentType
  ContentLength
  Custom(String)
} derive(Show, Hash, Eq)

///|
pub fn Header::to_string(self : Header) -> String {
  match self {
    Header::ContentType => "Content-Type"
    Header::ContentLength => "Content-Length"
    Header::Custom(s) => s
  }
}

///|
fn Header::from_string(s : String) -> Header {
  match s {
    "Content-Type" => Header::ContentType
    "Content-Length" => Header::ContentLength
    _ => Header::Custom(s)
  }
}

///|
type Headers Map[Header, String] derive(Default)

///|
fn Headers::new() -> Headers {
  Map::new()
}

///|
fn Headers::from_array(headers : Array[(Header, String)]) -> Headers {
  Map::from_array(headers)
}

///|
fn Headers::default_as_text() -> Headers {
  Headers::from_array(
    [(Header::ContentType, ContentType::TextPlain.to_string())],
  )
}

///|
fn Headers::default_as_json() -> Headers {
  Headers::from_array(
    [(Header::ContentType, ContentType::ApplicationJson.to_string())],
  )
}

///|
fn Headers::get(self : Headers, header : Header) -> String? {
  self._.get(header)
}

///|
fn Headers::set(self : Headers, header : Header, value : String) -> Unit {
  self._.set(header, value)
}

///|
pub fn Headers::merge(self : Headers, other : Headers) -> Unit {
  for k, v in other._ {
    self.set(k, v)
  }
}

///|
fn Headers::to_wasi(self : Headers) -> @types.Fields {
  @types.Fields::from_list(
    self._.iter()
    |> Iter::map(fn { (k, v) => (k.to_string(), @utils.encode_to_wasi(v)) })
    |> Iter::collect(),
  ).unwrap()
}

///|
fn Headers::from_wasi(fields : @types.Fields) -> Headers {
  Headers::from_array(
    fields
    .entries()
    .map(fn { (k, v) => (Header::from_string(k), @utils.decode_from_wasi(v)) }),
  )
}
