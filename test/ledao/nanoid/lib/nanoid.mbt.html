fn len_32(x: Int) -> Int {
    let mut n = 0
    let mut mut_x = x
    if mut_x >= 1 << 16 {
        mut_x = mut_x >> 16
        n = 16
    }
    if mut_x >= 1 << 8 {
        mut_x = mut_x >> 8
        n += 8
    }
    n +  len8tab[mut_x].to_int()
}

fn leading_zeros_32(x: Int) -> Int {
    32 - len_32(x)
}

fn generate_random_buffer(size: Int) -> Array[Byte] {
    let buffer = Array::make(size, b'\x00')
    let seed = b"ABCDEFGHIJKLMNOPQRSTUVWXYZ123466"
    let r = @random.new(seed~)
    let mut next_byte = 0
    let mut index = 0
    while index < size {
        next_byte = r.int(limit=256)
        buffer[index] = next_byte.to_byte()
        index = index + 1
    }
    buffer
}

fn format_string(alphabet: String, size: Int) -> String {
    let mask = (2 << (31 - leading_zeros_32(alphabet.length() -1 | 1))) - 1
    let step = ((1.6 * (mask.to_double() * size.to_double()) / (alphabet.length().to_double())).ceil()).to_int()
    let mut id = ""
    while true {
        let random_buffer = generate_random_buffer(step)
        let mut index = 0
        while index < step {
            let current_index = (random_buffer[index] & mask.to_byte()).to_int()
            if current_index < alphabet.length() {
                id = id + alphabet[current_index].to_string()
                if id.length() >= size {
                    return id
                }
            }
            index = index + 1
        }
    }
    id
}

fn generate_string(alphabet: String, size: Int) -> String {
    format_string(alphabet, size)
}

pub fn new(alphabet~: String = default_alphabet, size~: Int = default_size) -> String {
    generate_string(alphabet, size)
}
