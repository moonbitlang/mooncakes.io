///|
pub(readonly) struct Request {
  url : URL
  method : Method
  headers : Headers
  body : Body
  options : RequestOptions
}

///|
pub fn Request::new(
  url~ : URL,
  method~ : Method,
  headers~ : Headers = Headers::default_as_json(),
  body~ : Body = Body::empty(),
  options~ : RequestOptions = RequestOptions::default()
) -> Request {
  { url, method, headers, body, options }
}

///|
fn Request::to_wasi(self : Request) -> @types.OutgoingRequest {
  let req = @types.OutgoingRequest::outgoing_request(self.headers.to_wasi())
  req.set_method(self.method.to_wasi()) |> ignore
  req.set_scheme(Some(self.url.scheme.to_wasi())) |> ignore
  req.set_authority(Some(self.url.authority.to_string())) |> ignore
  req.set_path_with_query(Some(self.url.path_with_query())) |> ignore
  let stream = req.body().unwrap().write().unwrap()
  stream.blocking_write_and_flush(self.body.to_wasi()) |> ignore()
  req
}

///|
fn Request::from_wasi(req : @types.IncomingRequest) -> Request {
  let scheme = Scheme::from_wasi(req.scheme().unwrap())
  let authority = req.authority().unwrap()
  let url = URL::parse_path_with_query(
    scheme,
    authority,
    req.path_with_query().unwrap(),
  )
  let method = Method::from_wasi(req.method())
  let headers = Headers::from_wasi(req.headers())
  let content_length = @strconv.parse_int?(
      headers.get(Header::ContentLength).or(""),
    )
    .or(@int.max_value)
    .to_uint64()
  let stream = req.consume().unwrap().stream().unwrap()
  let buffer = @buffer.new()
  loop stream.blocking_read(content_length) {
    Ok(data) => {
      buffer.write_bytes(data)
      continue stream.blocking_read(content_length)
    }
    Err(_) => break
  }
  let body = Body::from_wasi(buffer.to_bytes())
  Request::new(url~, method~, headers~, body~)
}

///|
pub fn Request::send(self : Request) -> Response!RequestError {
  guard let Ok(future) = @outgoingHandler.handle(
    self.to_wasi(),
    self.options.to_wasi(),
  ) else {
    Err(err) => raise RequestError::WASIError(err)
  }
  future.subscribe().block()
  guard let Some(Ok(Ok(resp))) = future.get() else {
    Some(Ok(Err(err))) => raise RequestError::WASIError(err)
    _ =>
      raise RequestError::WASIError(
        @types.ErrorCode::InternalError(Some("unknown error")),
      )
  }
  Response::from_wasi(resp)
}
