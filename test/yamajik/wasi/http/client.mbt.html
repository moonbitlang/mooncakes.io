///|
struct Client {
  url : String
  headers : Headers
  options : RequestOptions
  middlewares : Middlewares
  logger : @log.Logger
}

///|
pub fn Client::new(
  url~ : String = "",
  headers~ : Headers = Headers::default_as_json(),
  options~ : RequestOptions = RequestOptions::default(),
  middlewares~ : Middlewares = Middlewares::new(),
  logger~ : @log.Logger = @log.Logger::new()
) -> Client {
  { url, headers, options, middlewares, logger }
}

///|
pub fn Client::request(
  self : Client,
  path~ : String,
  method~ : Method,
  headers~ : Headers = Headers::new(),
  body~ : Body = Body::empty(),
  options~ : RequestOptions = RequestOptions::default()
) -> Response! {
  let url = URL::parse!(self.url + path)
  headers.merge(self.headers)
  options.merge(self.options)
  let req = Request::new(url~, method~, headers~, body~, options~)
  let resp = self.middlewares.wrap(fn(req) { req.send!() })!(req)
  if resp.status_code.is_error() {
    self.logger.debug(
      "[\{req.method}] \{req.url.path} - \{resp.status_code} > \{resp.body}",
    )
  } else {
    self.logger.debug("[\{req.method}] \{req.url.path} - \{resp.status_code}")
  }
  resp
}

///|
pub fn Client::get(
  self : Client,
  path~ : String = "",
  headers~ : Headers = Headers::new(),
  body~ : Body = Body::empty(),
  options~ : RequestOptions = RequestOptions::default()
) -> Response! {
  self.request!(path~, method=Method::Get, headers~, body~, options~)
}
