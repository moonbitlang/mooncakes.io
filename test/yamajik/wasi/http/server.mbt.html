///|
struct Server {
  router : Router
  middlewares : Middlewares
  logger : @log.Logger
}

///|
pub fn Server::new(
  router~ : Router = Router::new(),
  middlewares~ : Middlewares = Middlewares::new(),
  logger~ : @log.Logger = @log.Logger::new()
) -> Server {
  { router, middlewares, logger }
}

///|
pub fn Server::on_handle(
  self : Server,
  path~ : String,
  method~ : Method,
  handler~ : Handler
) -> Unit {
  self.router.on_handle(path~, method~, handler=self.middlewares.wrap(handler))
}

///|
pub fn Server::router(self : Server, prefix : String) -> Router {
  self.router.sub(prefix)
}

///|
pub fn Server::serve(self : Server, p0 : Int, p1 : Int) -> Unit {
  let req = Request::from_wasi(@types.IncomingRequest::IncomingRequest(p0))
  let resp = try {
    self.handle!(req)
  } catch {
    e => e.to_response()
  }
  @types.ResponseOutparam::ResponseOutparam(p1).set(resp.to_wasi() |> Ok)
  if resp.status_code.is_error() {
    self.logger.debug(
      "[\{req.method}] \{req.url.path} - \{resp.status_code} > \{resp.body}",
    )
  } else {
    self.logger.debug("[\{req.method}] \{req.url.path} - \{resp.status_code}")
  }
}

///|
pub fn Server::handle(self : Server, req : Request) -> Response!HTTPError {
  let path = req.url.path.to_string()
  let method = req.method
  guard let Some(handler) = self.router.get_handler(path~, method~) else {
    None => raise HTTPError::NotFound({ "path": path })
  }
  try {
    handler!(req)
  } catch {
    e => raise HTTPError::from_error(e)
  }
}
