/// A stack of values of type `T`.
///
/// # Usage
///
/// ```
/// fn init {
///   // Create a new stack of integers
///   let stack : Stack[Int] = Stack.make()
/// 
///   // Push some values onto the stack
///   stack.push(10)
///   stack.push(20)
///   stack.push(30)
/// 
///   // Pop values from the stack
///   let value1 = stack.pop()
///   let value2 = stack.pop()
///   let value3 = stack.pop()
/// 
///   // Print the popped values
///   print(value1) // Output: 30
///   print(value2) // Output: 20
///   print(value3) // Output: 10
/// }
/// ```
///
struct Stack[A] {
  mut data : @immut/list.T[A]
}

/// Pushes a value onto the stack.
///
/// # Arguments
///
/// * `val` - The value to be pushed onto the stack.
///
pub fn push[T](self : Stack[T], val : T) -> Unit {
  self.data = Cons(val, self.data)
}

/// Pops a value from the stack.
///
/// # Returns
///
/// The value popped from the stack.
///
/// # Panics
///
/// Panics if the stack is empty.
///
pub fn pop[T](self : Stack[T]) -> T {
  match self.data {
    Nil => abort("Stack underflow")
    Cons(x,xs) => {
      self.data = xs
      x
    }
  }
}

/// Creates a new empty stack.
///
/// # Returns
///
/// A new empty stack.
///
pub fn Stack::make[T]() -> Stack[T] {
  { data: Nil }
}
