///|
pub(all) type! HTTPError {
  BadRequest(ToJson)
  Unauthorized(ToJson)
  Forbidden(ToJson)
  NotFound(ToJson)
  InternalServerError(ToJson)
  NotImplemented(ToJson)
  BadGateway(ToJson)
  ServiceUnavailable(ToJson)
  Other(UInt, ToJson)
}

///|
pub fn HTTPError::code(self : HTTPError) -> StatusCode {
  match self {
    BadRequest(_) => BadRequest
    Unauthorized(_) => Unauthorized
    Forbidden(_) => Forbidden
    NotFound(_) => NotFound
    InternalServerError(_) => InternalServerError
    NotImplemented(_) => NotImplemented
    BadGateway(_) => BadGateway
    ServiceUnavailable(_) => ServiceUnavailable
    Other(i, _) => Other(i)
  }
}

///|
pub fn HTTPError::json(self : HTTPError) -> Json {
  match self {
    BadRequest(data) => data.to_json()
    Unauthorized(data) => data.to_json()
    Forbidden(data) => data.to_json()
    NotFound(data) => data.to_json()
    InternalServerError(data) => data.to_json()
    NotImplemented(data) => data.to_json()
    BadGateway(data) => data.to_json()
    ServiceUnavailable(data) => data.to_json()
    Other(_, data) => data.to_json()
  }
}

///|
pub fn HTTPError::to_response(self : HTTPError) -> Response {
  Response::json(status_code=self.code(), data=self.json())
}

///|
pub fn HTTPError::from_response(resp : Response) -> HTTPError?! {
  let data = resp.body.json!()
  match resp.status_code {
    BadRequest => Some(BadRequest(data))
    Unauthorized => Some(Unauthorized(data))
    Forbidden => Some(Forbidden(data))
    NotFound => Some(NotFound(data))
    InternalServerError => Some(InternalServerError(data))
    NotImplemented => Some(NotImplemented(data))
    BadGateway => Some(BadGateway(data))
    ServiceUnavailable => Some(ServiceUnavailable(data))
    Other(i) => Some(Other(i, data))
    _ => None
  }
}

///|
pub fn HTTPError::from_error(err : Error) -> HTTPError {
  match err {
    HTTPError::BadRequest(_) as e => e
    HTTPError::Unauthorized(_) as e => e
    HTTPError::Forbidden(_) as e => e
    HTTPError::NotFound(_) as e => e
    HTTPError::InternalServerError(_) as e => e
    HTTPError::NotImplemented(_) as e => e
    HTTPError::BadGateway(_) as e => e
    HTTPError::ServiceUnavailable(_) as e => e
    HTTPError::Other(_) as e => e
    _ => InternalServerError({ "message": "unexpected error" })
  }
}
