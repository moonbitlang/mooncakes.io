///|
pub(readonly) struct HTTPServer {
  server : @http.Server
}

///|
pub fn HTTPServer::new(
  server~ : @http.Server = @http.Server::new()
) -> HTTPServer {
  { server, }
}

///|
pub fn HTTPServer::on_invoke[A : @json.FromJson, B : ToJson](
  self : HTTPServer,
  name~ : String,
  method~ : @http.Method = @http.Method::Post,
  handler~ : (A) -> B!
) -> Unit {
  let path = "/" + name.trim("/")
  self.server.on_handle(
    path~,
    method~,
    handler=self.json_handler_wrapper(handler),
  )
}

///|
pub fn HTTPServer::on_binding[A : @json.FromJson, B : ToJson](
  self : HTTPServer,
  name~ : String,
  method~ : @http.Method = @http.Method::Post,
  handler~ : (A) -> B!
) -> Unit {
  let path = "/" + name.trim("/")
  self.server.on_handle(
    path~,
    method~,
    handler=self.json_handler_wrapper(handler),
  )
}

///|
pub fn HTTPServer::serve(self : HTTPServer, p0 : Int, p1 : Int) -> Unit {
  self.server.serve(p0, p1)
}

///|
fn json_handler_wrapper[A : @json.FromJson, B : ToJson](
  self : HTTPServer,
  handler : (A) -> B!
) -> (@http.Request) -> @http.Response! {
  fn(req : @http.Request) -> @http.Response! {
    let data = try {
      @json.from_json!(req.body.json!())
    } catch {
      _ =>
        raise @http.HTTPError::BadRequest(
          { "message": "Invalid Json Request Body" },
        )
    }
    let return_data = try {
      handler!(data)
    } catch {
      DaprError::InvalidArgument(data) =>
        raise @http.HTTPError::BadRequest(data)
      e => raise e
    }
    @http.Response::json(data=return_data.to_json())
  }
}
