    UInt(x) => x().to_string()
    ArrayInt(x) => x().to_string()
    ArrayFloat(x) => x().to_string()
    ArrayDouble(x) => x().to_string()
    ArrayString(x) => x().to_string()

    ArrayChar(x) => x().to_string()
    ArrayInt64(x) => x().to_string()
    ArrayUInt64(x) => x().to_string()
    ArrayUInt(x) => x().to_string()
  }
}
