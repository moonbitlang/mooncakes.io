test "test_uint64" {
  let test_number : UInt64 = 1145141919810
  let tested : UInt64 = test_number
    |> uint64_to_byte_array
    |> byte_array_to_uint64_raw
  if test_number != tested {
    fail!("Not equal after bytes conversion!")
  }
}
