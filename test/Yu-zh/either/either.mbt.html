pub enum Either[L, R] {
  Left(L)
  Right(R)
}

pub fn is_left[L, R](self: Either[L, R]) -> Bool {
  match self {
    Either::Left(_) => true
    Either::Right(_) => false
  }
}

pub fn is_right[L, R](self: Either[L, R]) -> Bool {
  match self {
    Either::Left(_) => false
    Either::Right(_) => true
  }
}

pub fn left[L, R](self: Either[L, R]) -> Option[L] {
  match self {
    Either::Left(l) => Some(l)
    Either::Right(_) => None
  }
}

pub fn right[L, R](self: Either[L, R]) -> Option[R] {
  match self {
    Either::Left(_) => None
    Either::Right(r) => Some(r)
  }
}

pub fn flip[L, R](self: Either[L, R]) -> Either[R, L] {
  match self {
    Either::Left(l) => Either::Right(l)
    Either::Right(r) => Either::Left(r)
  }
}

pub fn map_left[L, R, L2](self: Either[L, R], f: (L) -> L2) -> Either[L2, R] {
  match self {
    Either::Left(l) => Either::Left(f(l))
    Either::Right(r) => Either::Right(r)
  }
}

pub fn map_right[L, R, R2](self: Either[L, R], f: (R) -> R2) -> Either[L, R2] {
  match self {
    Either::Left(l) => Either::Left(l)
    Either::Right(r) => Either::Right(f(r))
  }
}

pub fn map_either[L, R, L2, R2](self: Either[L, R], f: (L) -> L2, g: (R) -> R2) -> Either[L2, R2] {
  match self {
    Either::Left(l) => Either::Left(f(l))
    Either::Right(r) => Either::Right(g(r))
  }
}

pub fn either[L, R, T](self: Either[L, R], f: (L) -> T, g: (R) -> T) -> T {
  match self {
    Either::Left(l) => f(l)
    Either::Right(r) => g(r)
  }
}
