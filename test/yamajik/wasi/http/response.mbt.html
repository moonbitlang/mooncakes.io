///|
pub(readonly) struct Response {
  status_code : StatusCode
  headers : Headers
  body : Body
}

///|
pub fn Response::new(
  status_code~ : StatusCode = StatusCode::Ok,
  headers~ : Headers = Headers::new(),
  body~ : Body = Body::empty()
) -> Response {
  { status_code, headers, body }
}

///|
pub fn Response::text(
  status_code~ : StatusCode = StatusCode::Ok,
  headers~ : Headers = Headers::default_as_text(),
  data~ : String = ""
) -> Response {
  Response::new(status_code~, headers~, body=Body::new(data~))
}

///|
pub fn Response::json(
  status_code~ : StatusCode = StatusCode::Ok,
  headers~ : Headers = Headers::default_as_json(),
  data~ : Json = Json::Null
) -> Response {
  Response::new(
    status_code~,
    headers~,
    body=Body::new(data=@json.stringify(data)),
  )
}

///|
fn Response::to_wasi(self : Response) -> @types.OutgoingResponse {
  let response = @types.outgoing_response(self.headers.to_wasi())
  response
  .body()
  .unwrap()
  .write()
  .unwrap()
  .blocking_write_and_flush(self.body.to_wasi())
  |> ignore
  response.set_status_code(self.status_code.to_wasi()) |> ignore
  response
}

///|
fn Response::from_wasi(resp : @types.IncomingResponse) -> Response {
  let status_code = StatusCode::from_wasi(resp.status())
  let headers = Headers::from_wasi(resp.headers())
  let content_length = @strconv.parse_int?(
      headers.get(Header::ContentLength).or(""),
    )
    .or(@int.max_value)
    .to_uint64()
  let stream = resp.consume().unwrap().stream().unwrap()
  let buffer = @buffer.new()
  loop stream.blocking_read(content_length) {
    Ok(data) => {
      buffer.write_bytes(data)
      continue stream.blocking_read(content_length)
    }
    Err(_) => break
  }
  let body = Body::from_wasi(buffer.to_bytes())
  Response::new(status_code~, headers~, body~)
}
