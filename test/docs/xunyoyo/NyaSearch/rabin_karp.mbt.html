///|
enum Prework {
  Util(p~ : Array[Int64])
  Nil
}

///|
pub struct StringHash {
  n : Int
  mod : Int64
  base : Int64
  hash : Array[Int64]
  prework : Prework
}

///|
let empty_hash : StringHash = StringHash::{
  n: 0,
  mod: 998244353,
  base: 131,
  hash: Array::make(0, 0),
  prework: Prework::Nil,
}

///|
fn StringHash::new(
  mod~ : Int64 = 998244353,
  base~ : Int64 = 131,
  s : String
) -> StringHash {
  let n = s.length()
  if n == 0 {
    return empty_hash
  }
  let p = Array::make(n + 1, 0L)
  let h = Array::make(n + 1, 0L)
  p[0] = 1
  h[0] = 1
  for i = 1; i <= n; i = i + 1 {
    p[i] = p[i - 1] * base % mod
    h[i] = (h[i - 1] * base + (s[i - 1] - 'a' + 1).to_int64()) % mod
  }
  StringHash::{ n, mod, base, hash: h, prework: Util(p~) }
}

///|
fn StringHash::get_hash(self : StringHash, left : Int, right : Int) -> Int64 {
  let p = match self.prework {
    Util(p~) => p
    Nil => []
  }
  (
    self.hash[right + 1] -
    self.hash[left] * p[right - left + 1] % self.mod +
    self.mod
  ) %
  self.mod
}
