///|
type Memory

///|
extern "js" fn initializeMemory(min : Int, max : Int) -> Memory =
  #|(min, max) => {
  #|  return new WebAssembly.Memory({ initial: min, maximum: max, shared: true })
  #|}

///|
pub let memory : Memory = initializeMemory(1, 1)

///|
extern "js" fn memory_atomic_wait32_ffi(
  memory : Memory,
  addr : Int,
  expected : Int,
  timeout : Int64
) -> Int =
  #|(memory, addr, expected, timeout) => {
  #|  return Atomics.wait(new Int32Array(memory.buffer), addr, expected, timeout)
  #|}

///|
pub fn memory_atomic_wait32(addr : Int, expected : Int, timeout : Int64) -> Int {
  memory_atomic_wait32_ffi(memory, addr, expected, timeout)
}

///|
extern "js" fn i32_atomic_rmw_cmpxchg_ffi(
  memory : Memory,
  addr : Int,
  expected : Int,
  replacement : Int
) -> Int =
  #|(memory, addr, expected, replacement) => {
  #|  return Atomics.compareExchange(new Int32Array(memory.buffer), addr, expected, replacement)
  #|}

///|
pub fn i32_atomic_rmw_cmpxchg(
  addr : Int,
  expected : Int,
  replacement : Int
) -> Int {
  i32_atomic_rmw_cmpxchg_ffi(memory, addr, expected, replacement)
}

///|
pub extern "js" fn i32_atomic_store_ffi(
  memory : Memory,
  addr : Int,
  value : Int
) =
  #|(memory, addr, value) => {
  #|  Atomics.store(new Int32Array(memory.buffer), addr, value)
  #|}

///|
pub fn i32_atomic_store(addr : Int, value : Int) -> Unit {
  i32_atomic_store_ffi(memory, addr, value)
}

///|
extern "js" fn memory_atomic_notify_ffi(
  memory : Memory,
  addr : Int,
  count : Int
) =
  #|(memory, addr, count) => {
  #|  Atomics.notify(new Int32Array(memory.buffer), addr, count)
  #|}

///|
pub fn memory_atomic_notify(addr : Int, count : Int) -> Unit {
  memory_atomic_notify_ffi(memory, addr, count)
}
