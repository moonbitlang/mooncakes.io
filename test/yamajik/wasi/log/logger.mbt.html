///|
const NEWLINE = "\n"

///|
struct Logger {
  stream : @streams.OutputStream
}

///|
pub fn Logger::new() -> Logger {
  { stream: @stderr.get_stderr() }
}

///|
pub fn Logger::with_field(self : Logger, key : String, value : Show) -> Entry {
  let fields = Map::new()
  fields.set(key, value)
  Entry::new(self, fields~)
}

///|
pub fn Logger::with_fields(self : Logger, fields : Map[String, Show]) -> Entry {
  Entry::new(self, fields~)
}

///|
pub fn Logger::with_error(self : Logger, error : Error) -> Entry {
  self.with_field(FIELD_KEY_ERROR, error)
}

///|
fn Logger::log(self : Logger, message : String) -> Unit {
  self.stream.write(message |> @utils.encode_to_wasi()) |> ignore
  self.stream.blocking_write_and_flush(NEWLINE.to_bytes()) |> ignore
}

///|
pub fn Logger::debug(self : Logger, message : String) -> Unit {
  self.log("[DEBUG] \{message}")
}

///|
pub fn Logger::info(self : Logger, message : String) -> Unit {
  self.log("[INFO] \{message}")
}

///|
pub fn Logger::warn(self : Logger, message : String) -> Unit {
  self.log("[WARN] \{message}")
}

///|
pub fn Logger::error(self : Logger, message : String) -> Unit {
  self.log("[ERROR] \{message}")
}

///|
pub fn Logger::panic(self : Logger, message : String) -> Unit {
  self.log("[PANIC] \{message}")
  panic()
}
