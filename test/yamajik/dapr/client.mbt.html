///|
struct HTTPClient {
  client : @http.Client
  runtime_version : String
}

///|
pub fn HTTPClient::new(
  url~ : String = "",
  runtime_version~ : String = "v1.0"
) -> HTTPClient {
  let client = match url {
    "" => {
      let address = @env.get("DAPR_LISTEN_ADDRESS").as_string().or("127.0.0.1")
      let port = @env.get("DAPR_HTTP_PORT").as_string().or("50001")
      @http.Client::new(url="http://\{address}:\{port}")
    }
    s => @http.Client::new(url=s)
  }
  { client, runtime_version }
}

///|
pub fn HTTPClient::invoke[A : ToJson, B : @json.FromJson](
  self : HTTPClient,
  app~ : String,
  method~ : String,
  data~ : A
) -> B! {
  self.request!(
    path="/\{self.runtime_version}/invoke/\{app}/method/\{method}",
    data~,
  )
}

///|
fn HTTPClient::request[A : ToJson, B : @json.FromJson](
  self : HTTPClient,
  path~ : String,
  data~ : A
) -> B! {
  let resp = self.client.post!(
    path~,
    body=@http.Body::from_json(data=data.to_json()),
  )
  @json.from_json!(resp.body.json!())
}
